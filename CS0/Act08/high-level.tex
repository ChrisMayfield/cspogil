In addition to adding the numbers from 1 to 10, this program displays (prints) the result on the screen.

\begin{center}
\begin{minipage}[t]{186pt}
%\hrulefill

\textbf{Standard C} \\
(3rd Generation)

\begin{verbatim}
#include <stdio.h>

int main()
{
    int upper = 11;
    int sum = 1;
    int next = 2;

    while (next < upper)
    {
        sum = sum + next;
        next++;
    }

    printf("Sum = %d\n", sum);

    return 0;
}
\end{verbatim}
\end{minipage}
